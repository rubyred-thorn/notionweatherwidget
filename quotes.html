function displayQuote(index) { document.getElementById("quote").innerHTML = quotes[index]; } function prevQuote() { var currentIndex = quotes.indexOf(document.getElementById("quote").innerHTML); var prevIndex = (currentIndex - 1 + quotes.length) % quotes.length; displayQuote(prevIndex); } function nextQuote() { var currentIndex = quotes.indexOf(document.getElementById("quote").innerHTML); var nextIndex = (currentIndex + 1) % quotes.length; displayQuote(nextIndex); } window.onload = function () { displayQuote(0); document.getElementById("prev-button").addEventListener("click", prevQuote); document.getElementById("next-button").addEventListener("click", nextQuote); };
